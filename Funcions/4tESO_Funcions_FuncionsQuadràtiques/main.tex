\section*{Introducció}

Les \textbf{funcions quadràtiques} tenen la forma $f(x) = ax^2 + bx + c$ i la seva representació gràfica és una \textbf{paràbola}.

\vspace{0.3cm}

\noindent \textbf{Exemple visual:}
A la dreta tenim dues paràboles representades:
\begin{itemize}
    \item $f_1(x) = x^2 - 2x + 1$ (Línia contínua)
    \item $f_2(x) = -2x^2 + 8x - 5$ (Línia discontínua)
\end{itemize}

\noindent Per representar-les cal fer una taula de valors completa. 
\textbf{Exemple de càlcul per a $f_1(x) = x^2 - 2x + 1$:}
\begin{itemize}
    \item $x = -1 \rightarrow f_1(-1) = (-1)^2 - 2(-1) + 1 = 1 + 2 + 1 = 4$
    \item $x = 0 \phantom{-} \rightarrow f_1(0) = 0^2 - 2(0) + 1 = 1$
    \item $x = 1 \phantom{-} \rightarrow f_1(1) = 1^2 - 2(1) + 1 = 1 - 2 + 1 = 0$
\end{itemize}

    \section*{Propietats de la paràbola}
    
    \begin{itemize}
        \item \textbf{Creixement i decreixement:} Observa la gràfica d'esquerra a dreta.
        \item \textbf{Vèrtex:} És el punt on la paràbola dona la volta (el pic de la muntanya o el fons de la vall).
    \end{itemize}
    
    \noindent \textit{\textbf{Exemple:} La funció $f_1$ (pàgina 1) decreix fins a $x=1$ i creix a partir d'allà. El seu vèrtex és el punt $(1, 0)$, que és un mínim.}

    \section*{Taxa de variació mitjana (TVM)}
    
    La \textbf{TVM} mesura la pendent mitjana entre dos punts: 
    $ \text{TVM} = \frac{\Delta y}{\Delta x} = \frac{f(b)-f(a)}{b-a} $

    \begin{minipage}{0.55\textwidth}
        \textbf{Exemple gràfic:} \\
        En la funció $f_1(x)$ entre $x=1$ i $x=3$:
        \begin{itemize}
            \item Avancem 2 unitats a la dreta ($\Delta x = 2$).
            \item Pugem 4 unitats amunt ($\Delta y = 4$).
        \end{itemize}
        La TVM és $\frac{4}{2} = 2$.
    \end{minipage}
    \hfill
    \begin{minipage}{0.4\textwidth}
        \centering
        \begin{tikzpicture}
            \begin{axis}[
                width=6cm, height=5cm,
                axis lines = middle,
                grid = major,
                xmin=0, xmax=4,
                ymin=0, ymax=5,
                xtick={0,1,2,3,4},
                ytick={0,1,2,3,4}
            ]
                \addplot[domain=0:3.5, samples=100, color=black, thick] {x^2 - 2*x + 1};
                
                % Línies de cota per Delta x (baixades perquè es vegin)
                \draw[thick, dashed] (axis cs:1,0) -- (axis cs:1,-0.8);
                \draw[thick, dashed] (axis cs:3,0) -- (axis cs:3,-0.8);
                \draw[thick, <->] (axis cs:1,-0.5) -- (axis cs:3,-0.5) node[midway, fill=white] {$\Delta x=2$};
                
                % Delta y
                \draw[thick, dashed, ->] (axis cs:3,0) -- (axis cs:3,4) node[midway, right] {$\Delta y=4$};
                
                \addplot[mark=*] coordinates {(1,0) (3,4)};
            \end{axis}
        \end{tikzpicture}
    \end{minipage}

    \section*{Representació a partir del vèrtex}
    
    Per representar una paràbola és útil trobar primer el seu vèrtex $(x_v, y_v)$.
    \begin{itemize}
        \item Coordenada X: $x_{v} = \frac{-b}{2a}$
        \item Coordenada Y: Substituïm el valor trobat a la funció.
    \end{itemize}

    \newpage
    \noindent \textbf{Exemple:} $f_1(x) = x^2 - 2x + 1$ ($a=1, b=-2$)
    \begin{itemize}
        \item $x_v = \frac{-(-2)}{2\cdot 1} = \frac{2}{2} = 1$
        \item $y_v = f_1(1) = 1^2 - 2(1) + 1 = 0$
        \item \textbf{Vèrtex:} $(1, 0)$.
    \end{itemize}
