\begin{itemize}[nosep]

    \item \textbf{Obtús:} Mesura més de $90^\circ$ i menys de $180^\circ$ (és molt obert).
    \item \textbf{Pla:} Mesura exactament $180^\circ$ (és una línia totalment plana).
\end{itemize}

\begin{center}
\fcolorbox{black}{gray!10}{
\begin{minipage}{0.9\textwidth}
\textbf{Com classifiquem els triangles?} \\
\textbf{Segons els seus costats:}
\begin{itemize}[nosep]
    \item \textbf{Equilàter:} Els 3 costats són iguals.
    \item \textbf{Isòsceles:} Té 2 costats iguals i 1 de diferent.
    \item \textbf{Escalè:} Tots els 3 costats són diferents.
\end{itemize}
\vspace{0.2cm}
\textbf{Segons els seus angles:}
\begin{itemize}[nosep]
    \item \textbf{Acutangle:} Tots els 3 angles són aguts.
    \item \textbf{Rectangle:} Té un angle recte (de $90^\circ$).
    \item \textbf{Obtusangle:} Té un angle obtús.
\end{itemize}
\end{minipage}
}
\end{center}

\begin{center}

\begin{minipage}{0.9\textwidth}
Quan tenim un \textbf{Triangle Rectangle} (amb un angle de $90^\circ$):
\begin{itemize}[nosep]
    \item El costat més llarg es diu \textbf{HIPOTENUSA}. Sempre està \textbf{just al davant} de l'angle recte.
    \item Els altres dos costats més curts, que formen la "L", es diuen \textbf{CATETS}.
\end{itemize}
\end{minipage}
}

\begin{center}
\fcolorbox{black}{gray!10}{
\begin{minipage}{0.9\textwidth}
\textbf{La fórmula màgica (Teorema de Pitàgores):} \\
Ens serveix per calcular la hipotenusa si sabem quant mesuren els dos catets:
\begin{center}
    $\text{Hipotenusa}^2 = \text{Catet}^2 + \text{Catet}^2$
\end{center}
\end{minipage}
}
\end{center}

\textbf{Exemple resolt pas a pas:} \\
Volem trobar la hipotenusa ($h$) d'un triangle amb catets de $3\text{ cm}$ i $4\text{ cm}$.

\vspace{0.3cm}
\begin{minipage}{0.4\textwidth}
    \begin{center}
    \begin{tikzpicture}[scale=0.7]
        \draw[thick] (0,0) -- (4,0) node[midway, below]{$4\text{ cm}$} -- (0,3) node[midway, above right]{$h$} -- cycle node[midway, left]{$3\text{ cm}$};
        \draw (0.3,0) -- (0.3,0.3) -- (0,0.3);
    \end{tikzpicture}
    \end{center}
\end{minipage}
\begin{minipage}{0.55\textwidth}
    \begin{itemize}
        \item \textbf{1) Calcula el quadrat de la hipotenusa ($h^2$)} sumant els catets al quadrat:
        $$h^2 = 3^2 + 4^2 = 9 + 16 = 25$$
        \item \textbf{2) Fes l'arrel quadrada} per trobar la hipotenusa ($h$):
        $$h = \sqrt{25} = 5\text{ cm}$$
    \end{itemize}
\end{minipage}
