% --- EXERCICI 1: IGUALACIÓ ---
\section*{Exercici 1: Mètode d'Igualació}
Resoleu els següents sistemes mitjançant el mètode d'igualació.
Els primers apartats tenen la variable ja aïllada per facilitar el procés.

\begin{enumerate}[label=\alph*), itemsep=1.2cm]
    \begin{minipage}[t]{0.45\textwidth}
    % --- Bloc senzill (tipus 1a) ---
    \item $\begin{cases} y = 2x \\ y = x + 3 \end{cases}$
    \item $\begin{cases} x = 3y \\ x = y + 4 \end{cases}$
    \item $\begin{cases} y = 2x - 1 \\ y = x + 2 \end{cases}$
    \item $\begin{cases} x = 4y + 1 \\ x = 5y - 1 \end{cases}$
    \item $\begin{cases} y = x + 5 \\ y = 2x - 3 \end{cases}$
    \item $\begin{cases} x = 2y + 4 \\ x = 4y \end{cases}$
    \end{minipage}
    \begin{minipage}[t]{0.45\textwidth}
    % --- Més del bloc senzill ---
    \item $\begin{cases} y = 3x - 2 \\ y = 2x + 3 \end{cases}$
    \item $\begin{cases} x = y + 7 \\ x = 3y - 3 \end{cases}$
    
    % --- Bloc tipus 1b/1c (lleugera variació) ---
    \item $\begin{cases} x + y = 10 \\ x = 2y + 1 \end{cases}$
    \item $\begin{cases} 2x = y + 4 \\ y = 4x \end{cases}$
    \item $\begin{cases} y = 2x \\ y - x = 5 \end{cases}$
    \item $\begin{cases} x + 2y = 8 \\ x = y - 1 \end{cases}$
    \end{minipage}
\end{enumerate}

% --- EXERCICI 2: SUBSTITUCIÓ ---
\section*{Exercici 2: Mètode de Substitució}
Resoleu els següents sistemes mitjançant el mètode de substitució.
Aprofiteu la variable aïllada per substituir-la a l'altra equació.

\begin{enumerate}[label=\alph*), itemsep=1.2cm]
    \begin{minipage}[t]{0.45\textwidth}
    % --- Bloc senzill (Variable aïllada + variable sense coeficient a l'altra eq) ---
    \item $\begin{cases} x = y + 2 \\ x + y = 8 \end{cases}$
    \item $\begin{cases} y = 2x \\ x + y = 9 \end{cases}$
    \item $\begin{cases} x = 3y \\ x - y = 4 \end{cases}$
    \item $\begin{cases} y = x - 3 \\ x + y = 7 \end{cases}$
    \item $\begin{cases} x = 5 - y \\ x - y = 1 \end{cases}$
    \item $\begin{cases} y = 2x + 1 \\ y - x = 3 \end{cases}$
    \end{minipage}
    \begin{minipage}[t]{0.45\textwidth}
    % --- Bloc amb coeficient a la segona equació ---
    \item $\begin{cases} x = y + 1 \\ 2x + 3y = 12 \end{cases}$
    \item $\begin{cases} y = 3x \\ 2x + y = 10 \end{cases}$
    \item $\begin{cases} x = 2y - 1 \\ 3x - 2y = 5 \end{cases}$
    \item $\begin{cases} y = x + 2 \\ 2x - 3y = -9 \end{cases}$
    \item $\begin{cases} x = 4y \\ 2x - 5y = 6 \end{cases}$
    \item $\begin{cases} y = 5 - 2x \\ 3x + 2y = 9 \end{cases}$
    \end{minipage}
\end{enumerate}
