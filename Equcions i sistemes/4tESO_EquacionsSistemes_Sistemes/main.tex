% --- EXERCICI 1: IGUALACIÓ ---
\section*{Exercici 1: Mètode d'Igualació}
Resoleu els següents sistemes mitjançant el mètode d'igualació. Indiqueu clarament quina variable aïlleu.

\begin{enumerate}[label=\alph*), itemsep=1cm]
    \begin{minipage}[t]{0.45\textwidth}
    \item $\begin{cases} y = 2x - 3 \\ y = x + 1 \end{cases}$
    \item $\begin{cases} x = 3y + 2 \\ x = 5y + 4 \end{cases}$
    \item $\begin{cases} x + y = 5 \\ x - y = 1 \end{cases}$
    \item $\begin{cases} y + 2x = 8 \\ y - x = 2 \end{cases}$
    \item $\begin{cases} x + 3y = 10 \\ x - y = 2 \end{cases}$
    \item $\begin{cases} 2(x - 1) = y \\ 3x - y = 4 \end{cases}$
    \end{minipage}
    \begin{minipage}[t]{0.45\textwidth}
    \item $\begin{cases} x + 3y = 7 \\ 2x - y = 7 \end{cases}$
    \item $\begin{cases} 3(x + y) = 15 \\ 2x - y = 1 \end{cases}$
    \item $\begin{cases} 2x + y = 4 \\ 4x - y = 2 \end{cases}$
    \item $\begin{cases} 2x + 3y = 2 \\ x - y = 0 \end{cases}$
    \item $\begin{cases} 2x + y = 5 \\ 3x - 2y = 1 \end{cases}$
    \item $\begin{cases} \frac{x}{2} + y = 3 \\ x - \frac{y}{3} = 4 \end{cases}$
    \end{minipage}
\end{enumerate}

\section*{Exercici 2: Mètode de Substitució}
Resoleu els següents sistemes mitjançant el mètode de substitució. Es recomana aïllar la variable amb coeficient 1 o -1.

\begin{enumerate}[label=\alph*), itemsep=1cm]
    \begin{minipage}[t]{0.45\textwidth}
    \item $\begin{cases} x = 2y + 1 \\ 3x + y = 10 \end{cases}$
    \item $\begin{cases} y = 5 - x \\ 2x - 3y = 0 \end{cases}$
    \item $\begin{cases} x + 2y = 8 \\ 3x - y = 3 \end{cases}$
    \item $\begin{cases} 2x + y = 7 \\ 3(x + y) = 15 \end{cases}$
    \item $\begin{cases} 4x - y = 9 \\ 2x + 3y = 1 \end{cases}$
    \item $\begin{cases} 2x + 3y = 12 \\ x - 2y = -1 \end{cases}$
    \end{minipage}
    \begin{minipage}[t]{0.45\textwidth}
    \item $\begin{cases} 3x + 2y = 5 \\ 2x + y = 2 \end{cases}$
    \item $\begin{cases} 2x + y = 3 \\ 3x - 2y = 2 \end{cases}$
    \item $\begin{cases} 5x + y = 6 \\ 3x - 2y = 1 \end{cases}$
    \item $\begin{cases} x - 3y = -7 \\ 2x + 5y = 19 \end{cases}$
    \item $\begin{cases} 2(x-y) = 4 \\ 3x + y = 10 \end{cases}$
    \item $\begin{cases} \frac{x+y}{2} = 3 \\ 2x - y = 1 \end{cases}$
    \end{minipage}
\end{enumerate}

    \item $\begin{cases} 2x + 3y = 7 \\ 3x + 2y = 8 \end{cases}$

    \item $\begin{cases} x + y = 10 \\ 0.5x - 0.5y = 1 \end{cases}$
