% --- BREU EXPLICACIÓ (RECORDATORI) ---
\begin{center}
\fbox{\parbox{0.95\textwidth}{
\textbf{Recordatori de conceptes clau:}
\begin{itemize}[itemsep=2pt, leftmargin=1.5em]
    \item \textbf{Freqüència absoluta ($f_i$):} Quantitat de vegades que es repeteix una dada exacta.
    \item \textbf{Freqüència relativa ($h_i$):} Resultat de dividir $f_i$ pel total de dades ($N$). Per tenir el \textbf{\%}, multiplica aquest decimal per 100.
    \item \textbf{Freq. acumulades ($F_i$, $H_i$):} S'obtenen anant sumant les freqüències de les files anteriors.
\end{itemize}
}}
\end{center}
