% --- RECORDATORI ---
\noindent\fbox{
\begin{minipage}{\dimexpr\textwidth-2\fboxsep-2\fboxrule\relax}
\textbf{Recordatori:}
\begin{itemize}[noitemsep, topsep=0pt]
    \item \textbf{Mitjana ($\bar{x}$):} Suma de totes les dades dividida pel nombre total de dades ($N$).
    \item \textbf{Mediana ($Me$):} El valor central un cop ordenades les dades de menor a major.
    \item \textbf{Moda ($Mo$):} El valor o dada que més es repeteix.
    \item \textbf{Freqüència absoluta ($f_i$):} Nombre de vegades que apareix una dada.
    \item \textbf{Freqüència relativa ($h_i$ o $f_r$):} Quocient entre la freqüència absoluta i el total ($f_i / N$).
\end{itemize}
\end{minipage}}
